## Import

```ts
import { IconButton } from "@yamada-ui/react"
```

## Usage

```tsx
<IconButton icon={<FontAwesomeIcon icon={faPlus} />} />
```

### Change Variant

```tsx
<Wrap gap="md">
  <IconButton
    colorScheme="primary"
    variant="primary"
    icon={<FontAwesomeIcon icon={faPlus} />}
  />
  <IconButton
    colorScheme="secondary"
    variant="outline"
    icon={<FontAwesomeIcon icon={faMinus} />}
  />
  <IconButton
    colorScheme="warning"
    variant="ghost"
    icon={<FontAwesomeIcon icon={faSun} />}
  />
  <IconButton
    colorScheme="danger"
    variant="link"
    icon={<FontAwesomeIcon icon={faMoon} />}
  />
  <IconButton variant="unstyled" icon={<FontAwesomeIcon icon={faCheck} />} />
</Wrap>
```

### Change Size

```tsx
<Wrap gap="md">
  <IconButton
    colorScheme="primary"
    size="xs"
    icon={<FontAwesomeIcon icon={faPlus} />}
  />
  <IconButton
    colorScheme="secondary"
    size="sm"
    icon={<FontAwesomeIcon icon={faMinus} />}
  />
  <IconButton
    colorScheme="warning"
    size="md"
    icon={<FontAwesomeIcon icon={faSun} />}
  />
  <IconButton
    colorScheme="danger"
    size="lg"
    icon={<FontAwesomeIcon icon={faMoon} />}
  />
</Wrap>
```

### Disable

To disable, set `isDisabled` to `true`.

```tsx
<Wrap gap="md">
  <IconButton
    isDisabled
    colorScheme="primary"
    variant="primary"
    icon={<FontAwesomeIcon icon={faPlus} />}
  />
  <IconButton
    isDisabled
    colorScheme="secondary"
    variant="outline"
    icon={<FontAwesomeIcon icon={faMinus} />}
  />
  <IconButton
    isDisabled
    colorScheme="warning"
    variant="ghost"
    icon={<FontAwesomeIcon icon={faSun} />}
  />
  <IconButton
    isDisabled
    colorScheme="danger"
    variant="link"
    icon={<FontAwesomeIcon icon={faMoon} />}
  />
  <IconButton
    isDisabled
    variant="unstyled"
    icon={<FontAwesomeIcon icon={faCheck} />}
  />
</Wrap>
```
