## Import

```ts
import { Rotate } from "@yamada-ui/react"
```

## Usage

Set the elements to switch between as `from` and `to`.

```tsx functional=true
return (
  <Rotate from={<SunIcon fontSize="6xl" />} to={<MoonIcon fontSize="6xl" />} />
)
```

### Change the Angle

To change the angle, set a number (degrees) to `rotate`.

```tsx functional=true
return (
  <Rotate
    rotate={360}
    from={<SunIcon fontSize="6xl" />}
    to={<MoonIcon fontSize="6xl" />}
  />
)
```

### Change the Duration

To change the duration, set a number (seconds) to `duration`.

```tsx functional=true
return (
  <Rotate
    duration={0.6}
    from={<SunIcon fontSize="6xl" />}
    to={<MoonIcon fontSize="6xl" />}
  />
)
```

### Add a Delay

To delay the switch, set a number (seconds) to `delay`.

```tsx functional=true
return (
  <Rotate
    delay={1}
    from={<SunIcon fontSize="6xl" />}
    to={<MoonIcon fontSize="6xl" />}
  />
)
```

### Disable

To disable, set `disabled` to `true`.

```tsx functional=true
return (
  <Rotate
    disabled
    from={<SunIcon fontSize="6xl" />}
    to={<MoonIcon fontSize="6xl" />}
  />
)
```

### ReadOnly

To readyOnly, set `readOnly` to `true`.

```tsx functional=true
return (
  <Rotate
    readOnly
    from={<SunIcon fontSize="6xl" />}
    to={<MoonIcon fontSize="6xl" />}
  />
)
```

### Control

```tsx functional=true
const [value, onChange] = useState<RotateIdent>("to")

return (
  <Rotate
    value={value}
    onChange={onChange}
    from={<SunIcon fontSize="6xl" />}
    to={<MoonIcon fontSize="6xl" />}
  />
)
```
