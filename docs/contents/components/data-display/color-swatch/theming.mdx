## Theming

The `ColorSwatch` is a [multi part component](/styled-system/theming/component-styles#multi-part-components).

:::note
If you want to change the style of the component, please check [here](/styled-system/theming/customize-theme#changing-the-style-of-components).
:::

```ts
export const ColorSwatch: ComponentMultiStyle = {
  baseStyle: {
    container: {},
    overlay: { boxSize: "full" },
  },

  variants: {
    basic: {
      container: { rounded: "md" },
      overlay: { rounded: "md" },
    },
    rounded: {
      container: { rounded: "full" },
      overlay: { rounded: "full" },
    },
  },

  sizes: {
    sm: {
      container: { boxSize: "6" },
    },
    md: {
      container: { boxSize: "8" },
    },
    lg: {
      container: { boxSize: "10" },
    },
  },

  defaultProps: {
    variant: "basic",
    size: "md",
  },
}
```
