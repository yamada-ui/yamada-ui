## インポート

```ts
import { useTimeout } from "@yamada-ui/react"
```

## 使い方

```tsx functional=true
const [state, setState] = useState<number>(1)

useTimeout(() => setState((prev) => prev + 1), 3000)

return <Text>Current state: {state}</Text>
```
