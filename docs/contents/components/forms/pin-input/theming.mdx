## Theming

The `PinInput` is a [multi part component](/styled-system/theming/component-styles#multi-part-components).

:::note
`PinInput` inherits the style from [Input](/components/forms/input).
:::

:::note
If you want to change the style of the component, please check [here](/styled-system/theming/customize-theme#changing-the-style-of-components).
:::

```ts
export const PinInput: ComponentMultiStyle = mergeMultiStyle(Input, {
  baseStyle: {
    container: {
      gap: "sm",
    },
    field: {
      textAlign: "center",
    },
  },

  variants: {
    unstyled: {
      field: {
        h: "auto",
      },
    },
  },

  sizes: {
    xs: {
      field: {
        fontSize: "xs",
        boxSize: "6",
        rounded: "sm",
      },
    },
    sm: {
      field: {
        fontSize: "sm",
        boxSize: "8",
        rounded: "md",
      },
    },
    md: {
      field: {
        fontSize: "md",
        boxSize: "10",
        rounded: "md",
      },
    },
    lg: {
      field: {
        fontSize: "lg",
        boxSize: "12",
        rounded: "md",
      },
    },
  },
})({ omit: ["addon", "sizes"] })
```
