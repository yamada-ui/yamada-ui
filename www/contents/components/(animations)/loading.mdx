## Usage

:::code-group

```tsx [package]
import { Loading } from "@yamada-ui/react"
```

```tsx [alias]
import { Loading } from "@/components/ui"
```

```tsx [monorepo]
import { Loading } from "@workspaces/ui"
```

:::

```tsx
<Loading.Oval />
<Loading.Grid />
<Loading.Audio />
<Loading.Dots />
<Loading.Puff />
<Loading.Rings />
<Loading.Circles />
```
