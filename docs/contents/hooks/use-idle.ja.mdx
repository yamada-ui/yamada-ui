## インポート

```ts
import { useIdle } from "@yamada-ui/react"
```

## 使い方

```tsx functional=true
const idle = useIdle(2000)

return (
  <Box
    bg={idle ? "blue.500" : "red.500"}
    p="md"
    rounded="md"
    color="white"
    transitionProperty="all"
    transitionDuration="slower"
  >
    Current state: {idle ? "idle" : "not idle"}
  </Box>
)
```
