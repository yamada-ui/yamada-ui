```tsx preview functional client
const items = useMemo<Autocomplete.Item[]>(

    { label: "アグモン", value: "アグモン" },
    { label: "ガブモン", value: "ガブモン" },

    { label: "ピヨモン", value: "ピヨモン" },
    { label: "ゴマモン", value: "ゴマモン" },

    { label: "テイルモン", value: "テイルモン" },
  ],
  [],
)

return (
  <Autocomplete.Root
