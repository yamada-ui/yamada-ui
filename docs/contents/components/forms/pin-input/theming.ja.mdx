## テーマ

`PinInput`は、[複数パーツのコンポーネント](/styled-system/theming/component-styles#複数パーツのコンポーネント)です。

:::note
`PinInput`は、[Input](/components/forms/input)のスタイルを継承しています。
:::

:::note
コンポーネントのスタイルを変更したい場合は、[こちら](/styled-system/theming/customize-theme#コンポーネントのスタイルを変更する)をご覧ください。
:::

```ts
export const PinInput: ComponentMultiStyle<"PinInput"> = mergeMultiStyle(
  Input,
  {
    baseStyle: {
      container: {
        gap: "sm",
      },
      field: {
        textAlign: "center",
      },
    },

    sizes: {
      xs: {
        field: {
          boxSize: "6",
          fontSize: "xs",
          rounded: "sm",
        },
      },
      sm: {
        field: {
          boxSize: "8",
          fontSize: "sm",
          rounded: "md",
        },
      },
      md: {
        field: {
          boxSize: "10",
          fontSize: "md",
          rounded: "md",
        },
      },
      lg: {
        field: {
          boxSize: "12",
          fontSize: "lg",
          rounded: "md",
        },
      },
    },

    variants: {
      unstyled: {
        field: {
          h: "auto",
        },
      },
    },
  },
)({ omit: ["addon", "sizes"] })
```
