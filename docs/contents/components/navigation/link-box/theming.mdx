## Theming

The `LinkBox` is a [multi part component](/styled-system/theming/component-styles#multi-part-components).

:::note
If you want to change the style of the component, please check [here](/styled-system/theming/customize-theme#changing-the-style-of-components).
:::

```ts
export const LinkBox: ComponentMultiStyle<"LinkBox"> = {
  baseStyle: {
    container: {},
    overlay: {
      outline: 0,
      _focusVisible: {
        _before: {
          boxShadow: "outline",
        },
      },
    },
  },
}
```
