```tsx preview functional
      [
        "area",
        { dataKey: "tablet" },
      ] satisfies ComposedChart.ComposedProps<Data>,
      [
        "line",
        { dataKey: "mobile" },
      ] satisfies ComposedChart.ComposedProps<Data>,
    ]),
  [],
)
const data = useMemo(() => createCartesianChartData(), [])

return (
  <ComposedChart.Root
    data={data}
    series={series}
    withYAxis
    chartProps={{ margin: { right: 16 } }}
```
```tsx preview functional
    tooltipProps={{ labelFormatter: (value) => dayjs(value).format("MMM") }}
```
```tsx preview functional
    xAxisProps={{
      dataKey: "date",

)
```

## アクセシビリティ

### キーボード操作

| キー         | 説明                                                                 | 状態                 |
| ------------ | -------------------------------------------------------------------- | -------------------- |
| `ArrowRight` | 次のデータポイントにフォーカスを移動し、ツールチップを表示します。   | `withTooltip={true}` |
| `ArrowLeft`  | 前のデータポイントにフォーカスを移動し、ツールチップを表示します。   | `withTooltip={true}` |
| `Enter`      | フォーカスしているデータポイントのツールチップの表示を切り替えます。 | `withTooltip={true}` |

| -------------- | -------------------- | -------------------------------------- |
