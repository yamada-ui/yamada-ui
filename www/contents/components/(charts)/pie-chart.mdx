```tsx functional preview
interface Data {
  browser: string
  visits: number
  fill?: CSSProps["fill"]
}

const series = useMemo<PieChart.PieProps<Data>[]>(
  () => [{ dataKey: "visits", nameKey: "browser" }],
  [],

      {
        browser: "safari",

      },
      {
        browser: "other",
```

### Format

To format values and labels, use `formatter`.

```tsx preview functional
const series = useMemo<PieChart.PieProps<PolarChartData>[]>(
  () => [{ dataKey: "visits", nameKey: "browser" }],
  [],
)
const data = useMemo<PolarChartData[]>(
  () =>
    createPolarChartData().map((item, index) => ({
      ...item,
      fill: ["cyan", "green", "orange", "red", "blue", "gray"][index],
    })),
  [],
)

return (
  <PieChart.Root
    data={data}
    series={series}
    pieProps={{
      label: {
        formatter: (_, percent) => `${(percent * 100).toFixed(0)}%`,
      },
      labelLine: true,
      labelList: {
        color: "white",
        formatter: (value) => (isString(value) ? toTitleCase(value) : value),
      },
    }}
    tooltipProps={{
      formatter: (value, name = "") => [
        Number(value).toLocaleString(),
        toTitleCase(name),
      ],
    }}
  />
)
```

## Accessibility

### Keyboard Navigation

| Key          | Description                                                      | State                |
| ------------ | ---------------------------------------------------------------- | -------------------- |
| `ArrowRight` | Moves focus to the next data point and displays the tooltip.     | `withTooltip={true}` |
| `ArrowLeft`  | Moves focus to the previous data point and displays the tooltip. | `withTooltip={true}` |
| `Enter`      | Toggles the tooltip display for the focused data point.          | `withTooltip={true}` |

### ARIA Roles and Attributes

| Component | Role and Attribute   | Usage                                  |
| --------- | -------------------- | -------------------------------------- |
| `svg`     | `role="application"` | Indicates that this is an application. |
