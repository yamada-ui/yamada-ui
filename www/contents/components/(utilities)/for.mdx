title: For
description: "`For` is a component used to loop over an array and render a component for each item."
storybook: components-for--basic
source: components/for
---

```tsx preview functional
const items = useMemo(
  () => [
    { name: "竈門炭治郎", breathing: ["水の呼吸", "日の呼吸"] },
    { name: "我妻善逸", breathing: ["雷の呼吸"] },
    { name: "嘴平伊之助", breathing: ["獣の呼吸"] },
  ],
  [],
)

return (
  <VStack gap="md">
    <For each={items}>
      {({ name, breathing }, index) => (
        <Card.Root key={index}>
          <Card.Header>
            <Heading size="md">{name}</Heading>
          </Card.Header>

          <Card.Body pt="xs">
            <Text fontSize="sm" color="fg.muted">
              {breathing.join(", ")}
            </Text>
          </Card.Body>
        </Card.Root>
      )}
    </For>
  </VStack>
)
```

### Fallback

Use the `fallback` prop to render a fallback component when the array is empty or undefined.

```tsx preview
<VStack gap="md">
  <For
    each={[]}
    fallback={
      <EmptyState.Root
        description="There are no items to show"
        indicator={<BoxIcon />}
      />
    }
  >
    {(item, index) => <Text key={index}>{item}</Text>}
  </For>
</VStack>
```

### Filter

Use the `filter` prop to filter the array items.

```tsx preview functional
const items = useMemo(
  () => [
    { name: "竈門炭治郎", breathing: ["水の呼吸", "日の呼吸"] },
    { name: "我妻善逸", breathing: ["雷の呼吸"] },
    { name: "嘴平伊之助", breathing: ["獣の呼吸"] },
  ],
  [],
)

return (
  <VStack gap="md">
    <For each={items} filter={({ breathing }) => breathing.length == 1}>
      {({ name, breathing }, index) => (
        <Card.Root key={index}>
          <Card.Header>
            <Heading size="md">{name}</Heading>
          </Card.Header>

          <Card.Body pt="xs">
            <Text fontSize="sm" color="fg.muted">
              {breathing.join(", ")}
            </Text>
          </Card.Body>
        </Card.Root>
      )}
    </For>
  </VStack>
)
```

### Sort

Use the `sort` prop to sort the array items.

```tsx preview functional
const items = useMemo(
  () => [
    { name: "竈門炭治郎", breathing: ["水の呼吸", "日の呼吸"] },
    { name: "我妻善逸", breathing: ["雷の呼吸"] },
    { name: "嘴平伊之助", breathing: ["獣の呼吸"] },
  ],
  [],
)

return (
  <VStack gap="md">
    <For each={items} sort={(a, b) => a.breathing.length - b.breathing.length}>
      {({ name, breathing }, index) => (
        <Card.Root key={index}>
          <Card.Header>
            <Heading size="md">{name}</Heading>
          </Card.Header>

          <Card.Body pt="xs">
            <Text fontSize="sm" color="fg.muted">
              {breathing.join(", ")}
            </Text>
          </Card.Body>
        </Card.Root>
      )}
    </For>
  </VStack>
)
```
