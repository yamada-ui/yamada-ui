---
title: useHover
description: >-
  `useHover` is a custom hook that detects whether the pointer has moved over or
  away from an element.
package_name: "@yamada-ui/use-hover"
with_description: true
---

## Import

```ts
import { useHover } from "@yamada-ui/react"
```

## Usage

```tsx functional=true
const { hovered, ref } = useHover()

return (
  <Box ref={ref} w="full" bg="red.500" p="md" rounded="md" color="white">
    {hovered ? "I am hovered" : "Put mouse over me please"}
  </Box>
)
```
