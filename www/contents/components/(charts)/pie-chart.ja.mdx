interface Data {
  browser: string
  visits: number
  fill?: CSSProps["fill"]

const series = useMemo<PieChart.PieProps<Data>[]>(

### フォーマットする

フォーマットする場合は、`formatter`を使用します。

```tsx preview functional
const series = useMemo<PieChart.PieProps<Data>[]>(
  () => [{ dataKey: "visits", nameKey: "browser" }],
  [],
)
const data = useMemo<Data[]>(
  () =>
    createPolarChartData().map((item, index) => ({
      ...item,
      fill: ["cyan", "green", "orange", "red", "blue", "gray"][index],
    })),
  [],
)

return (
  <PieChart.Root
    data={data}
    series={series}
    pieProps={{
      label: {
        formatter: (_, percent) => `${(percent * 100).toFixed(0)}%`,
      },
      labelLine: true,
      labelList: {
        color: "white",
        formatter: (value) => (isString(value) ? toTitleCase(value) : value),
      },
    }}
    tooltipProps={{
      formatter: (value, name = "") => [
        Number(value).toLocaleString(),
        toTitleCase(name),
      ],
    }}
  />
)
```

## アクセシビリティ

### キーボード操作

| キー         | 説明                                                                 | 状態                 |
| ------------ | -------------------------------------------------------------------- | -------------------- |
| `ArrowRight` | 次のデータポイントにフォーカスを移動し、ツールチップを表示します。   | `withTooltip={true}` |
| `ArrowLeft`  | 前のデータポイントにフォーカスを移動し、ツールチップを表示します。   | `withTooltip={true}` |
| `Enter`      | フォーカスしているデータポイントのツールチップの表示を切り替えます。 | `withTooltip={true}` |

### ARIAロールと属性

| コンポーネント | ロールと属性         | 使い方                                 |
| -------------- | -------------------- | -------------------------------------- |
| `svg`          | `role="application"` | アプリケーションであることを示します。 |
