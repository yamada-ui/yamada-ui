## Theming

The `NativeSelect` is a [multi part component](/styled-system/theming/component-styles#multi-part-components).

:::note
`NativeSelect` inherits the style from [Input](/components/forms/input).
:::

:::note
If you want to change the style of the component, please check [here](/styled-system/theming/customize-theme#changing-the-style-of-components).
:::

```ts
export const NativeSelect: ComponentMultiStyle<"NativeSelect"> =
  mergeMultiStyle(Input, {
    baseStyle: {
      container: {},
      field: {
        "& > option, optgroup": {
          bg: ["white", "black"],
        },
        cursor: "pointer",
        pb: "px",
        pr: "8",
        _focus: {
          zIndex: "unset",
        },
        _readOnly: {
          pointerEvents: "none",
        },
      },
      icon: {
        color: ["blackAlpha.600", "whiteAlpha.700"],
        fontSize: "xl",
        outline: 0,
        py: "2",
        rounded: "md",
        w: "6",
        _disabled: {
          opacity: 0.4,
        },
      },
    },

    sizes: {
      xs: {
        icon: {
          fontSize: "lg",
          insetEnd: "1",
          pt: "3",
        },
      },
      sm: {
        icon: {
          insetEnd: "2",
        },
      },
      md: {
        icon: {
          insetEnd: "2",
        },
      },
      lg: {
        icon: {
          insetEnd: "2",
        },
      },
    },
  })({ omit: ["addon", "element"] })
```
