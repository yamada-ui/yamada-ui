    { label: "パルモン", value: "パルモン" },
    { label: "パタモン", value: "パタモン" },
    { label: "テイルモン", value: "テイルモン" },
  ],
  [],
)
return (
  <VStack>
    <For
      each={["scale", "block-end", "block-start", "inline-end", "inline-start"]}
