## インポート

```ts
import { useInterval } from "@yamada-ui/react"
```

## 使い方

```tsx functional=true
const [state, setState] = useState<number>(1)

useInterval(() => setState((prev) => prev + 1), 3000)

return <Text>Current state: {state}</Text>
```
