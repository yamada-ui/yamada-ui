## インポート

```ts
import { usePrevious } from "@yamada-ui/react"
```

## 使い方

```tsx functional=true
const [flg, { toggle }] = useBoolean()

const prevFlg = usePrevious(flg)

return (
  <>
    <Text>
      state: {String(flg)}, prev: {String(prevFlg)}
    </Text>

    <Button mt="md" onClick={toggle}>
      Click me to toggle the boolean value
    </Button>
  </>
)
```
