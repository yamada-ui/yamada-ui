```tsx preview functional client
const [device, setDevice] = useState("mobile");

useBreakpointEffect(breakpoint => {
  if (breakpoint === "sm") {
    setDevice("mobile");
  } else if (breakpoint === "md") {
    setDevice("tablet");
  } else {
    setDevice("desktop");
  }
}, []);

return <Text>The current device is "{device}"</Box>;
```
