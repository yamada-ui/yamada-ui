## インポート

```ts
import { Badge } from "@yamada-ui/react"
```

## 使い方

```tsx
<Badge>New</Badge>
```

### バリアントを変更する

```tsx
<Wrap gap="md">
  <Badge variant="subtle" colorScheme="primary">
    Primary
  </Badge>
  <Badge variant="solid" colorScheme="secondary">
    Secondary
  </Badge>
  <Badge variant="outline" colorScheme="success">
    Success
  </Badge>
</Wrap>
```
