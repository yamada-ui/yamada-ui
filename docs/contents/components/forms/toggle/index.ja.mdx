## インポート

```ts
import { Toggle, ToggleGroup } from "@yamada-ui/react"
```

## 使い方

```tsx
<Wrap gap="md">
  <Toggle icon={<FontAwesomeIcon icon={faBold} />} aria-label="Toggle bold" />

  <Toggle px="4">
    <FontAwesomeIcon icon={faItalic} />
    Italic
  </Toggle>
</Wrap>
```

### バリアントを変更する

```tsx
<Wrap gap="md">
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="primary"
    variant="subtle"
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="secondary"
    variant="solid"
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    variant="outline"
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    variant="unstyled"
  />
</Wrap>
```

### サイズを変更する

```tsx
<Wrap gap="md" alignItems="flex-start">
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="primary"
    size="xs"
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="secondary"
    size="sm"
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="warning"
    size="md"
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="danger"
    size="lg"
  />
</Wrap>
```

### 無効にする

無効にする場合は、`isDisabled`を`true`にします。

```tsx
<Wrap gap="md">
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="primary"
    variant="subtle"
    defaultIsSelected
    isDisabled
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="secondary"
    variant="solid"
    defaultIsSelected
    isDisabled
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    variant="outline"
    defaultIsSelected
    isDisabled
  />
</Wrap>
```

### 読み取り専用にする

読み取り専用にする場合は、`isReadOnly`を`true`に設定します。

```tsx
<Wrap gap="md">
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="primary"
    variant="subtle"
    defaultIsSelected
    isReadOnly
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="secondary"
    variant="solid"
    defaultIsSelected
    isReadOnly
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    variant="outline"
    defaultIsSelected
    isReadOnly
  />
</Wrap>
```

### 制御する

```tsx functional=true
const [isSelected, setIsSelected] = useState<boolean>(false)

return (
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    isSelected={isSelected}
    onChange={setIsSelected}
  />
)
```

### グループ化する

ボタンをグループ化する場合は、`ToggleGroup`を使用します。`ToggleGroup`は、子要素(`Toggle`)の`variant`, `size`, `colorScheme`, `isDisabled`, `isReadOnly`を一括で設定できます。

```tsx functional=true
const [value, setValue] = useState<string | undefined>(undefined)

return (
  <ToggleGroup value={value} onChange={setValue}>
    <Toggle
      value="bold"
      icon={<FontAwesomeIcon icon={faBold} />}
      aria-label="Toggle bold"
    />
    <Toggle
      value="italic"
      icon={<FontAwesomeIcon icon={faItalic} />}
      aria-label="Toggle italic"
    />
    <Toggle
      value="underline"
      icon={<FontAwesomeIcon icon={faUnderline} />}
      aria-label="Toggle underline"
    />
  </ToggleGroup>
)
```

複数選択の場合は、`value`または`defaultValue`を配列にします。

```tsx functional=true
const [value, setValue] = useState<string[]>([])

return (
  <ToggleGroup value={value} onChange={setValue}>
    <Toggle
      value="bold"
      icon={<FontAwesomeIcon icon={faBold} />}
      aria-label="Toggle bold"
    />
    <Toggle
      value="italic"
      icon={<FontAwesomeIcon icon={faItalic} />}
      aria-label="Toggle italic"
    />
    <Toggle
      value="underline"
      icon={<FontAwesomeIcon icon={faUnderline} />}
      aria-label="Toggle underline"
    />
  </ToggleGroup>
)
```
