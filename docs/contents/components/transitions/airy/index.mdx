## Import

```ts
import { Airy } from "@yamada-ui/react"
```

## Usage

Set the elements to switch between as `from` and `to`.

```tsx functional=true
return (
  <Airy
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```

### Change the Duration

To change the duration, set a numerical value (seconds) to `duration`.

```tsx functional=true
return (
  <Airy
    duration={0.6}
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```

### Add a Delay

If you want to delay the switch, set a numerical value (seconds) to `delay`.

```tsx functional=true
return (
  <Airy
    delay={1}
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```

### Disable

To disable, set `disabled` to `true`.

```tsx functional=true
return (
  <Airy
    disabled
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```

### ReadOnly

To readyOnly, set `readOnly` to `true`.

```tsx functional=true
return (
  <Airy
    readOnly
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```

### Control

```tsx functional=true
const [value, onChange] = useState<AiryIdent>("to")

return (
  <Airy
    value={value}
    onChange={onChange}
    from={<Text fontSize="2xl">ON</Text>}
    to={<Text fontSize="2xl">OFF</Text>}
  />
)
```
