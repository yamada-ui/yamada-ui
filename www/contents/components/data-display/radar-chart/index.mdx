## Import

<PackageManagers packageNameOrCommand="@yamada-ui/charts" />

:::note status=warning
`@yamada-ui/charts` is not included in `@yamada-ui/react`, so it needs to be installed separately.
:::

```ts
import { RadarChart } from "@yamada-ui/charts"
```

## Usage

:::note
`RadarChart` internally uses [Recharts](https://recharts.org/en-US/api/RadarChart).
:::

```tsx functional=true
const data = useMemo(
  () => [
    {
      name: "HP",
      ルギア: 106,
      ホウオウ: 106,
    },
    {
      name: "こうげき",
      ルギア: 90,
      ホウオウ: 130,
    },
    {
      name: "ぼうぎょ",
      ルギア: 130,
      ホウオウ: 90,
    },
    {
      name: "とくこう",
      ルギア: 90,
      ホウオウ: 110,
    },
    {
      name: "とくぼう",
      ルギア: 154,
      ホウオウ: 154,
    },
    {
      name: "すばやさ",
      ルギア: 110,
      ホウオウ: 90,
    },
  ],
  [],
)

const series: RadarProps[] = useMemo(
  () => [
    { dataKey: "ルギア", color: "blue.500" },
    { dataKey: "ホウオウ", color: "red.500" },
  ],
  [],
)

return <RadarChart data={data} series={series} dataKey="name" />
```

```tsx functional=true
)
```

### Format

To format tooltip values, use `valueFormatter`.
To format axis labels, use `polarAngleAxisTickFormatter`.
To format values on the radius axis, use `polarRadiusAxisTickFormatter`.

```tsx functional=true
const data = useMemo(
  () => [
    {
      name: "HP",
      ルギア: 106,
      ホウオウ: 106,
    },
    {
      name: "こうげき",
      ルギア: 90,
      ホウオウ: 130,
    },
    {
      name: "ぼうぎょ",
      ルギア: 130,
      ホウオウ: 90,
    },
    {
      name: "とくこう",
      ルギア: 90,
      ホウオウ: 110,
    },
    {
      name: "とくぼう",
      ルギア: 154,
      ホウオウ: 154,
    },
    {
      name: "すばやさ",
      ルギア: 110,
      ホウオウ: 90,
    },
  ],
  [],
)

const series: RadarProps[] = useMemo(
  () => [
    { dataKey: "ルギア", color: "blue.500" },
    { dataKey: "ホウオウ", color: "red.500" },
  ],
  [],
)

return (
  <VStack>
    <RadarChart
      data={data}
      series={series}
      dataKey="name"
      valueFormatter={(value) => `${value}P`}
    />
    <RadarChart
      data={data}
      series={series}
      dataKey="name"
      polarAngleAxisTickFormatter={(value) => `-${value}-`}
    />
    <RadarChart
      data={data}
      series={series}
      dataKey="name"
      withPolarRadiusAxis
      polarRadiusAxisTickFormatter={(value) => `${value}P`}
    />
  </VStack>
)
```
