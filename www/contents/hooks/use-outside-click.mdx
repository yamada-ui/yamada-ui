---
title: useOutsideClick
description: >-
  `useOutsideClick` is a custom hook that detects click events outside of an
  element.
package_name: "@yamada-ui/use-outside-click"
with_description: true
---

## Import

```ts
import { useOutsideClick } from "@yamada-ui/react"
```

## Usage

```tsx functional=true
const ref = useRef<HTMLDivElement>(null)
const [isOpen, setIsOpen] = useState(false)

useOutsideClick({
  ref,
  handler: () => setIsOpen(false),
})

return (
  <>
    {isOpen ? (
      <Center ref={ref} h="10" px="4" bg="danger" color="white" rounded="md">
        Hey, Click anywhere outside of me to close.
      </Center>
    ) : (
      <Button onClick={() => setIsOpen(true)}>Please Click</Button>
    )}
  </>
)
```
