### Composition

```tsx preview functional
const data = useMemo<Data[]>(() => createCartesianChartData(), [])

return (
  <AreaChart.Root data={data} chartProps={{ margin: { right: 16 } }}>
    <AreaChart.Area dataKey="desktop" dot stroke="colorScheme.900">
      <AreaChart.LabelList />
    </AreaChart.Area>
    <AreaChart.Area dataKey="tablet" stroke="colorScheme.600" />
    <AreaChart.Area dataKey="mobile" stroke="colorScheme.300" />
    <AreaChart.XAxis
      dataKey="date"
      tickFormatter={(value) => dayjs(value).format("MMM")}
    />
    <AreaChart.YAxis />
    <AreaChart.Grid />
    <AreaChart.Tooltip labelFormatter={(value) => dayjs(value).format("MMM")} />
  </AreaChart.Root>
)
```

```tsx preview functional
      },
    }}
    chartProps={{ margin: { right: 16 } }}
    tooltipProps={{
      formatter: (value) => `${(Number(value) / 1000).toFixed(1)}k`,
      labelFormatter: (value) => dayjs(value).format("MMM"),
    }}
    xAxisProps={{
      dataKey: "date",
      tickFormatter: (value) => dayjs(value).format("MMM"),
    }}
    yAxisProps={{
      domain: [0, 10000],
      tickFormatter: (value) => (value / 1000).toFixed(1),
      ticks: [0, 2500, 5000, 7500, 10000],
      unit: "k",
    }}
  />
)
```
