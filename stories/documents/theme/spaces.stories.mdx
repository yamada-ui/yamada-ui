## Default

<JSX>
  <Grid templateColumns="5rem 5rem 5rem 1fr" my="md" gap="md">
    <Text m="0">Name</Text>
    <Text m="0">Rem</Text>
    <Text m="0">Pixel</Text>
    <Box></Box>
    <Text m="0">px</Text>
    <Text m="0">0.0625rem</Text>
    <Text m="0">1px</Text>
    <Box maxW="1px" h="8xs" bg="green.500"></Box>
    {Object.entries(defaultTheme.spaces)
      .filter(([key]) => !isNaN(Number(key)))
      .sort(([a], [b]) => a - b)
      .map(([key, value]) => {
        return (
          <Fragment key={key}>
            <Text m="0">{key}</Text>
            <Text m="0">{value}</Text>
            <Text m="0">{`${parseFloat(value) * 16}px`}</Text>
            <Box maxW={value} h="8xs" bg="green.500"></Box>
          </Fragment>
        )
      })}
  </Grid>
</JSX>
