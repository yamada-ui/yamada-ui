## インポート

```ts
import { useWindowEvent } from "@yamada-ui/react"
```

## 使い方

```tsx functional=true
const [count, setCount] = useState(0)

useWindowEvent("click", () => {
  setCount(count + 1)
})

return <Text>{count}回クリックしました。</Text>
```
