## Usage

:::code-group

```tsx [package]
import { Kbd } from "@yamada-ui/react"
```

```tsx [alias]
import { Kbd } from "@/components/ui"
```

```tsx [monorepo]
import { Kbd } from "@workspace/ui"
```

:::

```tsx
<Text>
  <Kbd>Shift</Kbd> + <Kbd>Tab</Kbd>
</Text>
```

### Change Variant

```tsx preview
<Wrap gap="md">
  <For each={["solid", "subtle", "surface", "outline"]}>
    {(variant, index) => (
      <Kbd key={index} variant={variant}>
        Shift
      </Kbd>
    )}
  </For>
</Wrap>
```

### Change Size

```tsx preview
<Wrap gap="md">
  <For each={["sm", "md", "lg"]}>
    {(size, index) => (
      <Kbd key={index} size={size}>
        Tab
      </Kbd>
    )}
  </For>
</Wrap>
```

### Change Color Scheme

```tsx preview
<Wrap gap="md">
  <For each={["green", "pink"]}>
    {(colorScheme, index) => (
      <Kbd key={index} colorScheme={colorScheme}>
        Enter
      </Kbd>
    )}
  </For>
</Wrap>
```
