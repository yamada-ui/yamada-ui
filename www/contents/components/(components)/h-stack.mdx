```tsx preview
<HStack>
  {Array.from({ length: 4 }).map((_, index) => (
    <Box key={index} bg="bg.contrast" color="fg.contrast" p="md">
      Box
    </Box>
  ))}
</HStack>
```

## Usage

:::code-group

```tsx [package]
import { HStack } from "@yamada-ui/react"
```

```tsx [alias]
import { HStack } from "@/components/ui"
```

```tsx [monorepo]
import { HStack } from "@workspace/ui"
```

:::

```tsx
<HStack />
```

### Use spaces

```tsx preview
<HStack>
  <Box bg="bg.contrast" color="fg.contrast" p="md">
    Box
  </Box>

  <Spacer />

  <Box bg="bg.contrast" color="fg.contrast" p="md">
    Box
  </Box>
</HStack>
```
