## Alphas

  <Grid templateColumns='repeat(3, 1fr)' my='md' gap='md'>
    {Object.entries(defaultTheme.colors.whiteAlpha).map(([key, value]) => {
      return (
        <HStack key={key}>
          <Box bg={`whiteAlpha.${key}`} minW='12' minH='12' rounded='md' boxShadow='inner'></Box>
