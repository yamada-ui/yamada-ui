## Usage

:::code-group

```tsx [package]
import { Textarea, InputGroup } from "@yamada-ui/react"
```

```tsx [alias]
import { Textarea, InputGroup } from "@/components/ui"
```

```tsx [monorepo]
import { Textarea, InputGroup } from "@workspaces/ui"
```

:::

```tsx
<InputGroup.Root>
  <InputGroup.Addon />
  <Textarea />
  <InputGroup.Element />
</InputGroup.Root>
```

### Change Variants

```tsx preview
<VStack>
  <For each={["outline", "filled", "flushed"]}>
    {(variant) => (
      <Textarea
        key={variant}
        variant={variant}
        placeholder={toTitleCase(variant)}
      />
    )}
  </For>
</VStack>
```

### Change Size

```tsx preview
<VStack>

</VStack>
```

### Add Addons

```tsx preview
<VStack>
  <For each={["outline", "filled", "flushed"]}>
    {(variant) => (
      <InputGroup.Root key={variant} variant={variant}>
        <InputGroup.Addon>https://</InputGroup.Addon>
        <Textarea placeholder="Your site address" />
        <InputGroup.Addon>.com</InputGroup.Addon>
      </InputGroup.Root>
    )}
  </For>
</VStack>
```

### Change Border Color

To change the border color, set a color to `focusBorderColor` or `errorBorderColor`.

```tsx preview
<VStack>
  <Textarea focusBorderColor="green.500" placeholder="Custom border color" />
  <Textarea
    errorBorderColor="orange.500"
    invalid
    placeholder="Custom border color"
  />
</VStack>
```

### Customize Placeholder

```tsx preview
<Textarea
  placeholder="Custom placeholder"
  _placeholder={{ color: "green.500" }}
/>
```

### Change Resize Behavior

```tsx preview
<VStack>
  <For each={["block", "horizontal", "vertical", "none"]}>
    {(resize) => (
      <Textarea
        key={resize}
        placeholder={toTitleCase(resize)}
        resize={resize}
      />
    )}
  </For>
</VStack>
```

### Use Auto-resize

```tsx preview
<Textarea autosize placeholder="autosize" />
```

### Control Resize

```tsx preview functional client
const resizeRef = useRef<() => void>(null)
const onResize = () => {
  resizeRef.current?.()
}

return (
  <VStack>
    <Textarea placeholder="use resize" resizeRef={resizeRef} />
    <Button alignSelf="flex-end" onClick={onResize}>
      Resize
    </Button>
  </VStack>
)
```
