## Import

<PackageManagers packageNameOrCommand="@yamada-ui/charts" />

:::note status=warning
`@yamada-ui/charts` is not included in `@yamada-ui/react`, so it needs to be installed separately.
:::

```ts
import { PieChart } from "@yamada-ui/charts"
```

## Usage

:::note
`PieChart` internally uses [Recharts](https://recharts.org/en-US/api/PieChart).
:::

```tsx functional=true
const mewtwo: CellProps[] = useMemo(
  () => [
    {
      name: "HP",
      value: 106,
      color: "green.500",
    },
    {
      name: "こうげき",
      value: 110,
      color: "red.500",
    },
    {
      name: "ぼうぎょ",
      value: 90,
      color: "blue.500",
    },
    {
      name: "とくこう",
      value: 154,
      color: "purple.500",
    },
    {
      name: "とくぼう",
      value: 90,
      color: "orange.500",
    },
    {
      name: "すばやさ",
      value: 130,
      color: "cyan.500",
    },
  ],
  [],
)

return <PieChart data={mewtwo} />
```

To display the legend, set `withLegend` to `true`. To change the position of the legend, set `legendProps.verticalAlign` to `"bottom"` or similar.


### Adjust start angle, end angle

Set an angle to `startAngle` to adjust start angle or `endAngle` to adjust end angle.

:::note
The direction is anticlockwise with 3 o'clock as 0°. To go clockwise, set a negative value.
The default values are `90` for `startAngle` and `-270` for `endAngle` to go clockwise from 12 o'clock.
:::

```tsx functional=true
const mewtwo: CellProps[] = useMemo(
  () => [
    {
      name: "HP",
      value: 106,
      color: "green.500",
    },
    {
      name: "こうげき",
      value: 110,
      color: "red.500",
    },
    {
      name: "ぼうぎょ",
      value: 90,
      color: "blue.500",
    },
    {
      name: "とくこう",
      value: 154,
      color: "purple.500",
    },
    {
      name: "とくぼう",
      value: 90,
      color: "orange.500",
    },
    {
      name: "すばやさ",
      value: 130,
      color: "cyan.500",
    },
  ],
  [],
)

return <PieChart data={mewtwo} startAngle={90} endAngle={225} />
```

### Format

To format tooltip values, use `valueFormatter`.
To format label values, use `labelFormatter`.

```tsx functional=true
const mewtwo: CellProps[] = useMemo(
  () => [
    {
      name: "HP",
      value: 106,
      color: "green.500",
    },
    {
      name: "こうげき",
      value: 110,
      color: "red.500",
    },
    {
      name: "ぼうぎょ",
      value: 90,
      color: "blue.500",
    },
    {
      name: "とくこう",
      value: 154,
      color: "purple.500",
    },
    {
      name: "とくぼう",
      value: 90,
      color: "orange.500",
    },
    {
      name: "すばやさ",
      value: 130,
      color: "cyan.500",
    },
  ],
  [],
)

return (
  <VStack>
    <PieChart data={mewtwo} valueFormatter={(value) => `${value}P`} />
    <PieChart
      data={mewtwo}
      withLabels
      labelFormatter={(value) => `${value}P`}
    />
  </VStack>
)
```
