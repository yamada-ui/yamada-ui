## インポート

```ts
import { useResizeObserver } from "@yamada-ui/react"
```

## 使い方

```tsx functional=true
const [flg, { toggle }] = useBoolean()

const [ref, rect] = useResizeObserver<HTMLButtonElement>()

return (
  <>
    <Text>{JSON.stringify(rect)}</Text>

    <Button
      ref={ref}
      mt="md"
      onClick={toggle}
      w={flg ? "xl" : "sm"}
      h={flg ? "xl" : "sm"}
    >
      Click me to toggle the boolean value
    </Button>
  </>
)
```
