## Import

```ts
import { Toggle, ToggleGroup } from "@yamada-ui/react"
```

## Usage

```tsx
<Wrap gap="md">
  <Toggle icon={<FontAwesomeIcon icon={faBold} />} aria-label="Toggle bold" />

  <Toggle px="4">
    <FontAwesomeIcon icon={faItalic} />
    Italic
  </Toggle>
</Wrap>
```

### Change Variant

```tsx
<Wrap gap="md">
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="primary"
    variant="subtle"
  />

    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="secondary"
    variant="solid"
  />

    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    variant="outline"

  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}

    variant="unstyled"

```

### Disable

To disable, set `isDisabled` to `true`.

```tsx
<Wrap gap="md">
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="primary"
    variant="subtle"
    defaultIsSelected
    isDisabled
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="secondary"
    variant="solid"
    defaultIsSelected
    isDisabled
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    variant="outline"
    defaultIsSelected
    isDisabled
  />
</Wrap>
```

### Make Read-Only

To make read-only, set `isReadOnly` to `true`.

```tsx
<Wrap gap="md">
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="primary"
    variant="subtle"
    defaultIsSelected
    isReadOnly
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    colorScheme="secondary"
    variant="solid"
    defaultIsSelected
    isReadOnly
  />
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    variant="outline"
    defaultIsSelected
    isReadOnly
  />
</Wrap>
```

### Control

```tsx functional=true
const [isSelected, setIsSelected] = useState<boolean>(false)

return (
  <Toggle
    icon={<FontAwesomeIcon icon={faBold} />}
    aria-label="Toggle bold"
    isSelected={isSelected}
    onChange={setIsSelected}
  />
)
```

### Grouping

To group buttons, use `ToggleGroup`. `ToggleGroup` allows you to set `variant`, `size`, `colorScheme`, `isDisabled`, `isReadOnly` for child elements (`Toggle`) in bulk.

```tsx functional=true
const [value, setValue] = useState<string | undefined>(undefined)

return (
  <ToggleGroup value={value} onChange={setValue}>
    <Toggle
      value="bold"
      icon={<FontAwesomeIcon icon={faBold} />}
      aria-label="Toggle bold"
    />
    <Toggle
      value="italic"
      icon={<FontAwesomeIcon icon={faItalic} />}
      aria-label="Toggle italic"
    />
    <Toggle
      value="underline"
      icon={<FontAwesomeIcon icon={faUnderline} />}
      aria-label="Toggle underline"
    />
  </ToggleGroup>
)
```

For multiple selections, set `value` or `defaultValue` to an array.

```tsx functional=true
const [value, setValue] = useState<string[]>([])

return (
  <ToggleGroup value={value} onChange={setValue}>
    <Toggle
      value="bold"
      icon={<FontAwesomeIcon icon={faBold} />}
      aria-label="Toggle bold"
    />
    <Toggle
      value="italic"
      icon={<FontAwesomeIcon icon={faItalic} />}
      aria-label="Toggle italic"
    />
    <Toggle
      value="underline"
      icon={<FontAwesomeIcon icon={faUnderline} />}
      aria-label="Toggle underline"
    />
  </ToggleGroup>
)
```
