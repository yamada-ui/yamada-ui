## Usage

:::code-group

```tsx [package]
import { Steps } from "@yamada-ui/react"
```

```tsx [monorepo]
import { Steps } from "@workspaces/ui"
```

```tsx
<Steps.Root>
  <Steps.List>
    <Steps.Item>
      <Steps.Indicator />
      <Steps.Title />
      <Steps.Description />
      <Steps.Separator />
    </Steps.Item>
  </Steps.List>
  <Steps.Contents>
    <Steps.Content />
  </Steps.Contents>
  <Steps.CompletedContent />
  <Steps.PrevTrigger />
  <Steps.NextTrigger />
</Steps.Root>
```

### Use items

```tsx preview functional
const items = useMemo<Steps.Item[]>(
  () => [
    { description: "レッドリボン軍", title: "孫悟空少年編" },
    { description: "ピッコロ大魔王", title: "ピッコロ大魔王編" },
    { description: "ベジータ・ナッパ", title: "サイヤ人編" },
  ],
  [],
)

return (
  <Steps.Root items={items}>
    <Steps.List />
  </Steps.Root>
)
```

### Change variant

```tsx preview functional
const items = useMemo<Steps.Item[]>(
  () => [
    { description: "レッドリボン軍", title: "孫悟空少年編" },
    { description: "ピッコロ大魔王", title: "ピッコロ大魔王編" },
    { description: "ベジータ・ナッパ", title: "サイヤ人編" },
  ],
  [],
)

return (
  <VStack>
    <For each={["solid", "subtle"]}>
      {(variant) => (
        <Steps.Root
          key={variant}
          variant={variant}
          colorScheme="success"
          defaultIndex={1}
          items={items}
        >
          <Steps.List />
        </Steps.Root>
      )}
    </For>
  </VStack>
)
```
