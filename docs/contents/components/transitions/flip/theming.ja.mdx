## テーマ

`Flip`は、[複数パーツのコンポーネント](/styled-system/theming/component-styles#複数パーツのコンポーネント)です。

:::note
コンポーネントのスタイルを変更したい場合は、[こちら](/styled-system/theming/customize-theme#コンポーネントのスタイルを変更する)をご覧ください。
:::

```ts
export const Flip: ComponentMultiStyle<"Flip"> = {
  baseStyle: {
    container: {
      position: "relative",
      userSelect: "none",
      _disabled: {
        cursor: "not-allowed",
      },
      _readOnly: {
        cursor: "default",
      },
    },
    flipIdent: {
      backfaceVisibility: "hidden",
      position: "absolute",
      top: 0,
    },
    from: {},
    to: {},
  },

  variants: {},

  sizes: {},

  defaultProps: {},
}
```
