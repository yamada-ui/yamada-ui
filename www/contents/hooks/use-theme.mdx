```tsx preview functional client
const { theme } = useTheme()

return <Text lineClamp={3}>{JSON.stringify(theme)}</Text>
```

### Usage

:::code-group

```tsx [package]
import { useTheme } from "@yamada-ui/react"
```

```tsx [alias]
import { useTheme } from "@/components/ui"
```

```tsx [monorepo]
import { useTheme } from "@workspaces/ui"
```

:::

```tsx
const { themeScheme, changeThemeScheme } = useTheme()
```

:::note
If you want to learn more about the theme, please check [here](/docs/theming).
:::

#### Switching Themes

```tsx preview functional client
const { themeScheme, changeThemeScheme } = useTheme()

return (
  <>
    <Text>The current scheme is "{themeScheme}"</Text>

    <Wrap mt="md" gap="md">
      <Button onClick={() => changeThemeScheme("base")}>Base Theme</Button>

      <Button colorScheme="pink" onClick={() => changeThemeScheme("pink")}>
        Pink Theme
      </Button>

      <Button colorScheme="purple" onClick={() => changeThemeScheme("purple")}>
        Purple Theme
      </Button>

      <Button colorScheme="blue" onClick={() => changeThemeScheme("blue")}>
        Blue Theme
      </Button>
    </Wrap>
  </>
)
```

:::warning
In order to switch themes, you need to prepare multiple themes. For more details, please check [here](/docs/theming/switching-themes).
:::
