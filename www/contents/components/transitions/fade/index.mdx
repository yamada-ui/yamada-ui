## Import

```ts
import { Fade } from "@yamada-ui/react"
```

## Usage

To control the visibility, set `open`.

```tsx functional=true
const { open, onToggle } = useDisclosure()

return (
  <>
    <Button onClick={onToggle}>Please Click</Button>

    <Fade open={open}>
      <Box w="full" bg="orange.500" rounded="md" p="md" color="white" mt="md">
        クリリンのことか……クリリンのことかーーーっ！！！！！
      </Box>
    </Fade>
  </>
)
```

### Change the Duration

To change the duration, set a number (in seconds) to `duration`.

```tsx functional=true
const { open, onToggle } = useDisclosure()

return (
  <>
    <Button onClick={onToggle}>Please Click</Button>

    <Fade open={open} duration={0.4}>
      <Box w="full" bg="orange.500" rounded="md" p="md" color="white" mt="md">
        クリリンのことか……クリリンのことかーーーっ！！！！！
      </Box>
    </Fade>
  </>
)
```

### Unmount on Exit

To unmount the component when it is not visible, set `unmountOnExit` to `true`.

```tsx functional=true
const { open, onToggle } = useDisclosure()

return (
  <>
    <Button onClick={onToggle}>Please Click</Button>

    <Fade open={open} unmountOnExit>
      <Box w="full" bg="orange.500" rounded="md" p="md" color="white" mt="md">
        クリリンのことか……クリリンのことかーーーっ！！！！！
      </Box>
    </Fade>
  </>
)
```
