## 使い方

:::code-group

```tsx [package]
import { LinkBox } from "@yamada-ui/react"
```

```tsx [alias]
import { LinkBox } from "@/components/ui"
```

```tsx [monorepo]
import { LinkBox } from "@workspaces/ui"
```

:::

</LinkBox.Root>
