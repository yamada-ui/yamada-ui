## テーマ

`NativeSelect`は、[複数パーツのコンポーネント](/styled-system/theming/component-styles#複数パーツのコンポーネント)です。

:::note
`NativeSelect`は、[Input](/components/forms/input)のスタイルを継承しています。
:::

:::note
コンポーネントのスタイルを変更したい場合は、[こちら](/styled-system/theming/customize-theme#コンポーネントのスタイルを変更する)をご覧ください。
:::

```ts
export const NativeSelect: ComponentMultiStyle<"NativeSelect"> =
  mergeMultiStyle(Input, {
    baseStyle: {
      container: {},
      field: {
        "& > option, optgroup": {
          bg: ["white", "black"],
        },
        cursor: "pointer",
        pb: "px",
        pr: "8",
        _focus: {
          zIndex: "unset",
        },
        _readOnly: {
          pointerEvents: "none",
        },
      },
      icon: {
        color: ["blackAlpha.600", "whiteAlpha.700"],
        fontSize: "xl",
        outline: 0,
        py: "2",
        rounded: "md",
        w: "6",
        _disabled: {
          opacity: 0.4,
        },
      },
    },

    sizes: {
      xs: {
        icon: {
          fontSize: "lg",
          insetEnd: "1",
          pt: "3",
        },
      },
      sm: {
        icon: {
          insetEnd: "2",
        },
      },
      md: {
        icon: {
          insetEnd: "2",
        },
      },
      lg: {
        icon: {
          insetEnd: "2",
        },
      },
    },
  })({ omit: ["addon", "element"] })
```
