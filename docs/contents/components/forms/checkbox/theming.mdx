The `Checkbox` is a [multi part component](/styled-system/theming/component-style#multi-part-component).

:::note
If you want to change the style of the component, please check [here](/styled-system/theming/customize-theme#changing-the-style-of-components).
:::

export const Checkbox: ComponentMultiStyle = {
  baseStyle: {
    container: {
      _readOnly: { cursor: "auto" },

          bg: [`${c}.600`, `${c}.300`],
          borderColor: [`${c}.600`, `${c}.300`],

        _disabled: {

        },
      },
      _indeterminate: {
        bg: [`${c}.500`, `${c}.200`],
        borderColor: [`${c}.500`, `${c}.200`],
