## Theming

The `PinInput` is a [multi part component](/styled-system/theming/component-styles#multi-part-components).

:::note
`PinInput` inherits the style from [Input](/components/forms/input).
:::

:::note
If you want to change the style of the component, please check [here](/styled-system/theming/customize-theme#changing-the-style-of-components).
:::

```ts
export const PinInput: ComponentMultiStyle<"PinInput"> = mergeMultiStyle(
  Input,
  {
    baseStyle: {
      container: {
        gap: "sm",
      },
      field: {
        textAlign: "center",
      },
    },

    sizes: {
      xs: {
        field: {
          boxSize: "6",
          fontSize: "xs",
          rounded: "sm",
        },
      },
      sm: {
        field: {
          boxSize: "8",
          fontSize: "sm",
          rounded: "md",
        },
      },
      md: {
        field: {
          boxSize: "10",
          fontSize: "md",
          rounded: "md",
        },
      },
      lg: {
        field: {
          boxSize: "12",
          fontSize: "lg",
          rounded: "md",
        },
      },
    },

    variants: {
      unstyled: {
        field: {
          h: "auto",
        },
      },
    },
  },
)({ omit: ["addon", "sizes"] })
```
