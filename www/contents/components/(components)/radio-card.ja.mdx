```tsx preview

          <RadioCardGroup.Item.Label>
            {toTitleCase(variant)}
          </RadioCardGroup.Item.Label>

```

      value: "2",

### 方向を変更する

方向を変更する場合は、`orientation`に`"horizontal"`または`"vertical"`を設定します。デフォルトは`"horizontal"`です。

```tsx preview functional
const items = useMemo<RadioCardGroup.RootProps["items"]>(
  () => [
    {
      description:
        "パパスの息子より、ふたつ年上のしっかり者の幼なじみ。彼と一緒に、レヌール城のお化け退治に繰り出した。",
      label: "ビアンカ",
      value: "1",
    },
    {
      description:
        "大富豪ルドマンの娘で、デボラの妹。ちょっぴり天然なところもある、清楚で可憐なお嬢様。",
      label: "フローラ",
      value: "2",
    },
    {
      description:
        "なんと、この私が、好きと申すか！？そ、それはいかん！もう１度、考えてみなさい。",
      label: "ルドマン",
      value: "3",
    },
  ],
  [],
)

return (
  <VStack>
    <For each={["horizontal", "vertical"]}>
      {(orientation) => (
        <RadioCardGroup.Root
          key={orientation}
          items={items}
          orientation={orientation}
          defaultValue="1"
        />
      )}
    </For>
  </VStack>
)
```

### 無効化する

無効化する場合は、`disabled`を`true`に設定します。

```tsx preview
<VStack>
  <For each={["subtle", "surface", "outline"]}>
    {(variant) => (
      <RadioCardGroup.Root
        key={variant}
        variant={variant}
        defaultValue="1"
        disabled
      >
        <RadioCardGroup.Item.Root value="1">
          <RadioCardGroup.Item.Label>
            {toTitleCase(variant)}
          </RadioCardGroup.Item.Label>
        </RadioCardGroup.Item.Root>
        <RadioCardGroup.Item.Root value="2">
          <RadioCardGroup.Item.Label>No checked</RadioCardGroup.Item.Label>
        </RadioCardGroup.Item.Root>
      </RadioCardGroup.Root>

</VStack>
```
