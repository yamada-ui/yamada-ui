## テーマ

`Loading`は、[単一パーツのコンポーネント](/styled-system/theming/component-styles#単一パーツのコンポーネント)です。

:::note
コンポーネントのスタイルを変更したい場合は、[こちら](/styled-system/theming/customize-theme#コンポーネントのスタイルを変更する)をご覧ください。
:::

```ts
export const Loading: ComponentStyle<"Loading"> = {
  baseStyle: ({ colorScheme: c }) => ({
    color: [`${c}.500`, `${c}.600`],
  }),

  defaultProps: {
    colorScheme: "primary",
    size: "1em",
    variant: "oval",
  },
}
```
